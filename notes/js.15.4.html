5 ways of deep copy

Rest and Spread operator (...)

            let obj1={
                x:100,
                y:200
            }

            let obj2={...obj1} Using spread operator

Before

console.log("before -",obj1)
console.log("before -",obj2)

Output

before - {x: 100, y: 200}
before - {x: 100, y: 200}
                
            obj1.z=300;

After

console.log("after -",obj1)
console.log("after -",obj2)

Output

before - {x: 100, y: 200, z:300}
before - {x: 100, y: 200}

Loop (for-in)

            let obj1 = {
                x:100,
                y:200
            }
            let obj2 = {};
            
            for(let key in obj1){
                obj2[key]=obj1[key]
            }

Before

console.log("before -",obj1)
console.log("before -",obj2)

Output

before - {x: 100, y: 200}
before - {x: 100, y: 200}
                
            obj1.z=300;

After

Output

before - {x: 100, y: 200}
