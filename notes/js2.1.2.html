Prime Number Check Write a function is Prime(n) that checks whether a given number n is prime. Return true if the number is prime, otherwise return false.

function checkprime(num){
    if(num==1){
        return false
    }
    let nof = 0;
    for(let div=2;div*div<=num;div++){
        if(num%div==0){
            nof++;
            break;
        }
    }
    if(nof>0){
        return false
    }
    else{
        return true
    }
}

function fibonacci(num){
    let arr = [];
    let fd = 0;
    let sd = 1;
    if(num==1){
        arr.push(fd);
    }
    else if(num==2){
        arr.push(fd,sd);
    }
    else{
        arr.push(fd,sd);
        for(let i=1;i<=num-2;i++){
            let dig = fd + sd;
            arr.push(dig);
            fd = sd;
            sd = dig
        }
    }
    return arr;
}

Factorial Calculation Write a function factorial(n) that returns the factorial of a given number n. Example: factorial(5) should return 120 factorial(0) should return 1

function factorial(num){
    if(num==0){
        return 1
    }
    else{
        let dig = 1;
        for(let i=1;i<=num;i++){
            dig = dig * i;
        }
        return dig
    }   
}

removeDuplicates ([1, 2, 3, 1, 2, 4]);

// Expected output: [1, 2, 3, 4]

function removeDuplicates(array){
    for(let i=0;i < array.length;i++){
        for(let j=i+1;j < array.length;j++){
            if(array[i]==array[j]){
                array.splice(j,1)
            }
        }
    }
    return array
    
}
let array = [1,2,3,1,2,4]

        if(max < array[i]){
            max=array[i]

function isPalindrome(str){
    let rstr = "";

    for (let i = str.length - 1; i >= 0; i--) {
        rstr = rstr + str[i];
    }
    if(str==rstr){
        return true
    }
    else{
        return false
    }
}
