Functions (objects)

Suppose we need to make 3 objects with the same keys

            let emp1={
                name:"vikas",
                age:28,
                salary:10000,
                task:function(){
                    log("teaching)
                }
            }
            
            let emp2={
                name:"manas",
                age:18,
                salary:5000,
                task:function(){
                    log("office boy)
                }
            }
            
            let emp3={
                name:"saviur",
                age:30,
                salary:10000,
                task:function(){
                    log("operation)
                }
            }

Here the dry principle got affected so we will use function constructor (where constructor means new thing or new object)

Syntax

            function Employee(){
                this.name = "sparsh";
                this.age = 24;
                this.salary = 10000;
                this.task = function(){
                    console.log("student")
                }
            }

            let obj = new Employee()
            console.log(obj)

here name, age, salary, task are Keys and this means to bind or insert the value to key. For example- insert the "sparsh" to name key

Dynamic Function

            function Employee(name,age,salary,task){
                this.name = name;
                this.age = age;
                this.salary = salary;
                this.task = task
            }

            let obj = new Employee("sparsh",24,10000,function(){console.log("student")})
            console.log(obj)

            Output

            Employee {
                name: 'sparsh',
                age: 24,
                salary: 10000,
                task: [Function (anonymous)]
              }

            Run the task function
            obj.task()  //accessing the object properties by putting .keyname
            Output
            student
